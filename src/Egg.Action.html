data Action :: Nat -> ([Nat], Nat) -> Nat -> Nat -> Type where
    -- maxed out + locked + can't afford
    -- ResaerchError + Can't Afford
    AResearch
        :: ResearchIx tiers epic a
        -> Action eggs '(tiers, epic) habs vehicles
    -- Regression + Can't afford
    AHab
        :: Fin N4
        -> Finite habs
        -> Action eggs '(tiers, epic) habs vehicles
    -- Invalid slot + Regression + Can't afford
    -- aka VehicleError + can't afford
    AVehicle
        :: Integer
        -> Finite vehicles
        -> Action eggs '(tiers, epic) habs vehicles
    -- no error
    AWatchVideo
        :: Action eggs '(tiers, epic) habs vehicles
    -- Regression + not unlocked
    AEggUpgrade
        :: Finite eggs
        -> Action eggs '(tiers, epic) habs vehicles
    -- no error
    APrestige
        :: Action eggs '(tiers, epic) habs vehicles
